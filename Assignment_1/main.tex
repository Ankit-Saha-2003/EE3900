	\textbf{Problem 3.1.(e)} Determine the $Z$-transform, including the region of convergence, for the following sequence
	\begin{align}
		x(n) = \delta(n-1)
	\end{align}
	
	\solution 
	The $Z$-transform of a sequence $x(n)$ is defined as
	\begin{align}
		X(z) = \z{x(n)} = \sum_{n=-\infty}^\infty x(n) z^{-n}
	\end{align}
	
	The sequence is given by
	\begin{align}
		\delta(n-1) = 
		\begin{cases}
			1 & n = 1 \\
			0 & n \ne 1
		\end{cases}
	\end{align}
	
	Thus
	\begin{align}
		X(z) &= \sum_{n=-\infty}^\infty \delta(n-1) z^{-n} \\
		&= z^{-1}
	\end{align}
	
	Since there is only one non-zero term in the sequence, the sum will converge for any non-zero $z$. Thus, the region of convergence of this $Z$-transform is
	\begin{align}
		\abs{z} > 0
	\end{align}
	
	Therefore
	\begin{align}
		X(z) = z^{-1} \qquad \abs{z} > 0
	\end{align}
